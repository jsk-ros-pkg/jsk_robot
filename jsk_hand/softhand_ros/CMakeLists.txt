project(softhand_ros)

    actionlib_msgs
    message_generation
)

)

)

    actionlib_msgs
    message_runtime
)

  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
  USE_SOURCE_PERMISSIONS
)
